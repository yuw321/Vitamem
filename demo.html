import { createVitamem, InMemoryAdapter } from 'vitamem';

// Session 1: first check-in
const mem = await createVitamem({ llm, storage: new InMemoryAdapter() });
const thread = await mem.createThread({ userId: 'sarah' });

await mem.chat({ threadId: thread.id, message: "I've had Type 2 diabetes for 3 years" });
await mem.chat({ threadId: thread.id, message: "Taking metformin 1000mg twice a day" });
await mem.chat({ threadId: thread.id, message: "My last A1C was 7.4, goal is under 7.0" });

// End session: 1 LLM call extracts facts, then ~6 embedding calls (not 1 per message)
await mem.triggerDormantTransition(thread.id);

// Session 2: return visit
const memories = await mem.retrieve({
  userId: 'sarah',
  query: 'diabetes medications',
  limit: 10,
});
// [ { content: "Has Type 2 diabetes (3 years)", source: "confirmed" }, ... ]

const thread2 = await mem.createThread({ userId: 'sarah' });
await mem.chat({ threadId: thread2.id, message: "Hey, I'm back!" });
// AI: "Welcome back! Last time we talked about your A1C goal of 7.0..."
